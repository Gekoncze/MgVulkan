package cz.mg.vulkan.flags;

import cz.mg.vulkan.jna.flags.OLD_NAME;
import cz.mg.vulkan.jna.utilities.ToString;


public class NEW_NAME {
PROPERTIES

    public final OLD_NAME flags;

    public NEW_NAME(){
        this(0);
    }

    public NEW_NAME(int flag){
        this(new OLD_NAME(flag));
    }

    public NEW_NAME(OLD_NAME flags) {
        this.flags = flags;
    }

    public boolean isSet(int flag){
        return (flags.value & flag) != 0;
    }

    public void set(int flag){
        flags.value |= flag;
    }

    @Override
    public String toString() {
        return ToString.flagsToStringNoLabels(flags.value, OLD_NAME.class);
    }
}
