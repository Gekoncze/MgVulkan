package cz.mg.vulkan.enums;

import cz.mg.vulkan.jna.enums.OLD_NAME;
import cz.mg.vulkan.jna.utilities.ToString;


public class NEW_NAME {
PROPERTIES

    public final OLD_NAME enums;

    public NEW_NAME() {
        this(0);
    }

    public NEW_NAME(int value) {
        this(new OLD_NAME(value));
    }

    public NEW_NAME(OLD_NAME enums) {
        this.enums = enums;
    }

    public boolean is(int value){
        return this.enums.value == value;
    }

    public void set(int value){
        this.enums.value = value;
    }

    @Override
    public String toString() {
        return ToString.constantToString(enums.value, OLD_NAME.class);
    }
}
